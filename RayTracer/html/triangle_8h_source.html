108  Vector pvec = Vector::cross(ray.dir, edge2);
109  /* if determinant is near zero, ray lies in plane of triangle */

111  if (det > -EPSILON && det < EPSILON)
112  return -1;

118  if (u < 0.0f || u > 1.0f)
